本地搜索的实现
发布于: 2022-7-7   更新于: 2022-7-7   收录于: 0和1
本文字数: 1963   阅读时长: 4 分钟   阅读量:

无需配置的搜索功能它不香吗！

起因

配置 Algolia 的过程中，屡屡失败，对于新手来说，真心感到配置有些复杂。而且对于个人博客来说，也不太需要多么强大的搜索功能。那不如自己写一个吧。

实现原理

  1. 提取保存所有文章信息
  2. 绑定触发事件
  3. 内容处理
  4. 元素和样式

提取保存文章信息

如果只用 js 似乎只能获取当前页面的信息，并不能直接获取站点所有文章的信息，这个问题困扰了我一段时间。之后尝试过用 hugo 生成的 index.xml 来解决，呃，解决是能解决，就是结果太草率了，xml 里面提供的信息不够充分，导致搜索结果不准确，但好事是这为后面问题的解决提供了一点思路。既然是信息不足，那我们可以添加嘛，要加就得学 xml，学了小段时间后，便意识到这个方法成功的几率不太高，还复杂，就放弃了。

后来在翻看 Hugo 文档的时候，才发现原来 Hugo 本身是能够获取站点所有文章信息的（恍然大悟，看文档的重要性！！！），只需要3个变量： .Title 、 .PublishDate 和 .Plain 。然后在 body 标签里添加以下代码：

<!-- 创建空数组 -->
{{ $arrPosts := slice }}
<!-- 遍历每一篇文章 -->
{{ range .Site.RegularPages }}
    <!-- 提取文章信息：标题、发布日期和内容 -->
    {{ $thePost := dict "title" .Title "pubDate" (.PublishDate.Format "2006-01-02") "plain" .Plain "link" .Permalink }}
    <!-- 将文章信息添加到之前创建的数组中 -->
    {{ $arrPosts = $arrPosts | append $thePost }}
{{ end }}

<!-- 将得到的文章信息保存，以便 js 处理 -->
<script type="text/javascript">
    var postsCount = {{ $postsCount }};
    var arrPosts = {{ $arrPosts }};
</script>

// 主函数
function search() {
    clearPosts();
    var key = document.getElementById("search-key").value;
    var postCount = 0;

    if (!key) {
        $(".stip").html('~什么也没找到，');
        $(".at-bottom").hide();
        $(".sbody").show();
        return;
    }

    for (var i=0; i<postsCount; i++) {
        var postTitle = arrPosts[i].title;
        var postPubDate = arrPosts[i].pubDate;
        var postPlain = arrPosts[i].plain;
        var link = arrPosts[i].link;
        var keyIndex = postPlain.indexOf(key);
        
        // 匹配关键字
        if (keyIndex >= 0) {
            var postMark = toMark(postPlain, key);
            postCount ++;

            if (postMark) {
                addItem(hlHtml(postTitle, key), postPubDate, hlHtml(postMark, key), link);
            }

        }
        
    }
    if (postCount == 0) {
        $(".stip").html('~什么也没找到，');
    } else {
        $(".stip").html('找到 ' + postCount + ' 条结果，');
        $(".sbody-1").append('<div class="at-bottom">~~~ 已经到底啦</div>');
    }
    $(".sbody").show();
}

// 截取文字
function toMark(oPlain, key) {
    var kIdx = oPlain.indexOf(key);
    var kLen = key.length;
    var beginIdx = kIdx;
    var postMark_l = '';
    var postMark_r = oPlain.slice(kIdx + kLen, kIdx + kLen + 201);
    
    while ((beginIdx > 0) && (oPlain[beginIdx-1] != ',') && (oPlain[beginIdx-1] != '.') && (oPlain[beginIdx-1] != '，') && (oPlain[beginIdx-1] != '。')) {
        beginIdx -= 1;
        postMark_l = oPlain[beginIdx] + postMark_l;
    }

    if (postMark_l == key) {
        return;
    }

    return postMark_l + key + postMark_r;
     
}

// 高亮关键字
function hlHtml(oMark, key) {
    var text = oMark;
    var newMark = '';
    var keyIdx = text.indexOf(key);
    var keyHtml = "<span class=\"key-hl\">" + key + "</span>";
    while (keyIdx >= 0) {
        newMark = newMark + text.slice(0, keyIdx) + keyHtml;
        text = text.slice(keyIdx + key.length);
        keyIdx = text.indexOf(key);
    }
    return newMark + text;
}

// 向页面添加元素
function addItem(title, pubDate, mark, link) {
    var pHtml = "<a href=\"" + link + "\" target=\"_blank\" class=\"post\" >" +
            "<div class=\"post-header\">" +
            "<h4 class=\"post-title\">" + title + "</h4>" +
            "<div class=\"post-time\">" + pubDate + "</div>" +
            "</div>" +
            "<div class=\"post-mark\">" + mark + "</div>" +
            "</a>"
    var div = document.createElement("div");
    div.innerHTML = pHtml;
    div.setAttribute("class", "post-root");
    document.getElementsByClassName('sbody-1')[0].appendChild(div);
}

// 初始化
function clearPosts() {
    $(".sbody").hide();
    $(".post-root, .at-bottom").remove();
}
